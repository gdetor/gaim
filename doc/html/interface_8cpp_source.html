171  ga_pms.population_size = population_size;
172  ga_pms.num_offsprings = n_offsprings;
173  if (n_offsprings >= population_size) {
174  printf("Number of offsprings cannot exceed population size!\n");
175  exit(-1);
176  }
177  ga_pms.genome_size = genome_size;

191  pr_pms.print_bsf = log_bsf;

195

199  im_pms.pick_method = pickup_method;
200  im_pms.replace_method = replace_method;
201  im_pms.is_im_enabled = is_im_enabled;
202  im_pms.adj_list_fname = im_graph_fname;
203 
204  print_parameters(ga_pms, pr_pms, im_pms);
205  if (im_pms.is_im_enabled) {
206  IM island_model(&im_pms, &ga_pms);
207  printf("Optimizing using Island Model!\n");
208  for (size_t i = 0; i < im_pms.num_islands; ++i) {
209  island_model.island[i].fitness = func;
210  }
211  island_model.run_islands(&im_pms, &pr_pms);
212  } else {
213  if (ga_pms.runs == 1) {
214  printf("Optimizing using a single GA!\n");
215  GA gen_alg(&ga_pms);
216  gen_alg.fitness = func;
217  gen_alg.evolve(ga_pms.generations, 0, &pr_pms);
218  res.bsf = gen_alg.get_bsf();
219  res.average_fitness = gen_alg.get_average_fitness();
220  res.genome = gen_alg.get_best_genome();
221  } else if (ga_pms.runs > 1) {
222  printf("Running %d independent GAs!\n", ga_pms.runs);
223  independent_runs(&ga_pms, &pr_pms);
224  } else {
225  printf("Negative number of runs is illegal!\n");
226  exit(-1);
227  }
228  }
229  return res;
230 }
