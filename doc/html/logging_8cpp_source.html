55 std::string center(const std::string s, const int w) {
56  std::stringstream ss, spaces;
57  int padding = w - s.size();
58  for(int i=0; i<padding/2; ++i)
59  spaces << " ";
60  ss << spaces.str() << s << spaces.str();
61  if(padding > 0 && padding % 2 !=0)
62  ss << " ";
63  return ss.str();
64 }
