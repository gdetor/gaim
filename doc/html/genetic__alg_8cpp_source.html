39 GA::GA(ga_parameter_s *ga_pms)
40 {
41  alpha = ga_pms->a; // Lower bound for genes [a, b]
42  beta = ga_pms->b; // Upper bound for genes [a, b]
43 
44  std::random_device rd;
45  std::mt19937 gen(rd());
46  std::uniform_real_distribution<> genes(alpha, beta);
47 
48  mu = ga_pms->population_size; // Population size
49  if (mu < 2) {
50  std::cout << "Population size is smaller than 2!" << std::endl;
51  std::cout << "Impossible to have sexual reproduction!" << std::endl;
52  exit(-1);
53  }
54  lambda = ga_pms->num_offsprings; // Number of offspring
55  replace_perc = ga_pms->num_replacement; // Number of individuals being replaced
56  if (replace_perc > lambda) {
57  std::cout << "Generation replacement dimension is larger than \
58  offsprings population size!" << std::endl;
59  exit(-1);
60  }
61  genome_size = ga_pms->genome_size; // Genome size
62 
63  fitness = sphere; // Define the cost function (example -> sphere)
64 
65  // Initialize the population vector
66  for (size_t i = 0; i < mu; ++i) {
67  population.push_back(individual_s());
68  population[i].id = i; // Unique ID
69  population[i].fitness = -10000; // Fitness value
70  population[i].is_selected = false;
71  for (size_t j = 0; j < genome_size; ++j) {
72  population[i].genome.push_back(genes(gen)); // Genome
73  }
74  }
75  sorted_population = population; // Sort the population based on fitness
76 
77  // Initialize the offsprings vector
78  for (size_t i = 0; i < lambda; ++i) {
79  offsprings.push_back(individual_s());
80  offsprings[i].id = i;
81  offsprings[i].fitness = -10000;
82  for (size_t j = 0; j < genome_size; ++j) {
83  offsprings[i].genome.push_back(genes(gen));
84  }
85  }
86 
87  // FIXME to complete the following
88  if (ga_pms->universal_clipping == false) {
89  for (size_t i = 0; i < mu; ++i) {
90  population[i].lower_limit;
91  population[i].upper_limit;
92  }
93  }
94 }

259 void GA::next_generation(size_t perc=3)
260 {
261  std::sort(population.begin(), population.end(), compare_fitness);
262  std::sort(offsprings.begin(), offsprings.end(), compare_fitness);
263  if (perc > lambda) {
264  std::cout << "Percentage of offspring is larger than the\
265  available number of offspring!" << std::endl;
266  exit(-1);
267  }
268  for (size_t i = 0, j = lambda-1; i < perc; ++i, --j) {
269  population[i].genome = offsprings[j].genome;
270  population[i].fitness = offsprings[j].fitness;
271  }
272 }

329 void GA::run_one_generation(void)
330 {
331  /* int order; */
332  /* size_t xover; */
333  individual_s parent1, parent2;
334  std::vector<individual_s> parents;
335  std::vector<REAL_> child;
336 
337  // Iteration index
338  static size_t iter_index = 0;
339 
340  // Evaluate fitness of each individual
341  evaluation(population);
342 
343  // Sort the entire population based on fitness
344  sort_population();
345 
346  // Bookkeeping
347  best_individual = sorted_population[sorted_population.size()-1];
348  hfi.push_back(best_individual.fitness); // Best fitness
349  bsf_genome = best_individual.genome; // Best so far genome
350  lfi.push_back(sorted_population[0].fitness); // Lowest fitness
351 
352  // Average fitness across population
353  REAL_ acc = std::accumulate(population.begin(),
354  population.end(),
355  0,
356  average_fitness);
357  fit_avg.push_back(acc / static_cast<REAL_>(mu));
358 
359  bsf.push_back(best_individual.fitness);
360 
361  // Generate new offspring
362  for(size_t i = 0; i < lambda-1; ++i) {
363  // Parents selection
364  parents = selection(2, 2, false);
365  parent1 = parents[0];
366  parent2 = parents[1];
367  /* parent1 = parents[i]; */
368  /* parent2 = parents[i+1]; */
369 
370  // Reset the selection flags
371  // This has to be placed outside the loop in case the selection method
372  // is placed outside the loop
373  reset_selection_flags();
374 
375  // Crossover
376  child = crossover(parent1.genome, parent2.genome);
377  // child = order_one_crossover(parent1.genome, parent2.genome);
378 
379  // Mutation
380  /* REAL_ tmp = 85*pow((1 - 1/5000), iter_index) + 0.5; */
381  /* child = mutation(child, 0.5, tmp); */
382  child = mutation(child);
383  // child = swap_mutation(child);
384 
385  // Append the offspring genome list
386  offsprings[i].genome = child;
387  }
388  // Evaluate offspring fitness
389  evaluation(offsprings);
390 
391  // Integrate offspring in the initial population
392  next_generation(replace_perc);
393 
394  // Increase iteration index
395  iter_index++;
396 }

414 void GA::evolve(size_t generations,
415  size_t unique_id,
416  pr_parameter_s *pms)
417 {

434  }
435 }
