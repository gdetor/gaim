39 GA::GA(ga_parameter_s *ga_pms)
40 {
41  REAL_ bound_val;
42  // alpha and beta are vectors
43  alpha = ga_pms->a; // Lower bound for genes [a, b]
44  beta = ga_pms->b; // Upper bound for genes [a, b]
45 
46  // Validate alpha and beta paremeters
47  if ((alpha.size() != ga_pms->genome_size) && \
48  (beta.size() != ga_pms->genome_size)) {
49  std::cout << "Genome limits [a, b] size is not correct!" << std::endl;
50  std::cout << "Size of a and b = genome size!" << std::endl;
51  exit(-1);
52  }
53 
54  // Initialize random number generator
55  std::random_device rd;
56  std::mt19937 gen(rd());
57  std::uniform_real_distribution<> genes(alpha[0], beta[0]);
58 
59  // Assign and validate mu values
60  mu = ga_pms->population_size; // Population size
61  if (mu < 2) {
62  std::cout << "Population size is smaller than 2!" << std::endl;
63  std::cout << "Impossible to have sexual reproduction!" << std::endl;
64  exit(-1);
65  }
66  // Assign and validate lambda values
67  lambda = ga_pms->num_offsprings; // Number of offspring
68  replace_perc = ga_pms->num_replacement; // Number of individuals being replaced
69  if (replace_perc > lambda) {
70  std::cout << "Generation replacement dimension is larger than \
71  offsprings population size!" << std::endl;
72  exit(-1);
73  }
74  genome_size = ga_pms->genome_size; // Genome size
75 
76  fitness = sphere; // Define the cost function (example -> sphere)
77 
78  // Initialize selection method
79  selection_method = ga_pms->sel_pms.selection_method;
80  bias = ga_pms->sel_pms.bias;
81  num_parents = ga_pms->sel_pms.num_parents;
82  lower_bound = ga_pms->sel_pms.lower_bound;
83  k = ga_pms->sel_pms.k;
84  replace = ga_pms->sel_pms.replace;
85 
86  select_selection_method();
87 
88  // Initialize crossover method
89  crossover_method = ga_pms->cross_pms.crossover_method;
90  select_crossover_method();
91 
92  // Initialize mutation method
93  mutation_method = ga_pms->mut_pms.mutation_method;
94  mutation_rate = ga_pms->mut_pms.mutation_rate;
95  variance = ga_pms->mut_pms.variance;
96  low_bound = ga_pms->mut_pms.low_bound;
97  up_bound = ga_pms->mut_pms.up_bound;
98  time = ga_pms->mut_pms.time;
99  order = ga_pms->mut_pms.order;
100  is_real = ga_pms->mut_pms.is_real;
101 
102  select_mutation_method();
103 
104  // Initialize the population vector
105  for (size_t i = 0; i < mu; ++i) {
106  population.push_back(individual_s());
107  population[i].id = i; // Unique ID
108  population[i].fitness = -10000; // Fitness value
109  population[i].is_selected = false;
110  for (size_t j = 0; j < genome_size; ++j) {
111  population[i].genome.push_back(genes(gen)); // Genome
112  genes = std::uniform_real_distribution<>(alpha[j], beta[j]);
113  }
114  }
115  sorted_population = population; // Sort the population based on fitness
116 
117  // Initialize the offsprings vector
118  for (size_t i = 0; i < lambda; ++i) {
119  offsprings.push_back(individual_s());
120  offsprings[i].id = i;
121  offsprings[i].fitness = -10000;
122  for (size_t j = 0; j < genome_size; ++j) {
123  offsprings[i].genome.push_back(genes(gen));
124  genes = std::uniform_real_distribution<>(alpha[j], beta[j]);
125  }
126  }
127 
128  // corresponding file (see config file)
129  if (ga_pms->clipping == "file") {
130  auto ifile = std::fstream(ga_pms->clipping_fname, std::ios::in);
131  if (!ifile) {
132  std::cout << "Unable to open file " << ga_pms->clipping_fname\
133  << std::endl;
134  exit(1);
135  }
136 
137  for (size_t i = 0; i < mu; ++i) {
138  for (size_t j = 0; j < ga_pms->genome_size; ++j) {
139  ifile >> bound_val;
140  population[i].lower_limit.push_back(bound_val);
141  }
142  for (size_t j = 0; j < ga_pms->genome_size; ++j) {
143  ifile >> bound_val;
144  population[i].upper_limit.push_back(bound_val);
145  }
146  }
147  } else if (ga_pms->clipping == "individual") {
148  for (size_t i = 0; i < mu; ++i) {
149  for (size_t j = 0; j < ga_pms->genome_size; ++j) {
150  population[i].lower_limit.push_back(alpha[j]);
151  }
152  for (size_t j = 0; j < ga_pms->genome_size; ++j) {
153  population[i].upper_limit.push_back(beta[j]);
154  }
155  }
156  } else {
157  for (size_t i = 0; i < mu; ++i) {
158  for (size_t j = 0; j < ga_pms->genome_size; ++j) {
159  population[i].lower_limit.push_back(alpha[0]);
160  }
161  for (size_t j = 0; j < ga_pms->genome_size; ++j) {
162  population[i].upper_limit.push_back(beta[0]);
163  }
164  }
165  }
166 }

224 void GA::next_generation(size_t perc=3)
225 {
226  std::sort(population.begin(), population.end(), compare_fitness);
227  std::sort(offsprings.begin(), offsprings.end(), compare_fitness);
228  if (perc > lambda) {
229  std::cout << "Percentage of offspring is larger than the\
230  available number of offspring!" << std::endl;
231  exit(-1);
232  }
233  for (size_t i = 0, j = lambda-1; i < perc; ++i, --j) {
234  population[i].genome = offsprings[j].genome;
235  population[i].fitness = offsprings[j].fitness;
236  }
237 }

271 void GA::clip_genome()
272 {
273  // Alternative we could use replace_if
274  for (auto &ind : population) {
275  for (size_t i = 0; i < ind.genome.size(); ++i) {
276  if (ind.genome[i] > ind.upper_limit[i]) { ind.genome[i] = ind.upper_limit[i];}
277  if (ind.genome[i] < ind.lower_limit[i]) { ind.genome[i] = ind.lower_limit[i];}
278  }
279  }

295 void GA::run_one_generation(void)
296 {
297  /* int order; */
298  /* size_t xover; */
299  individual_s parent1, parent2;
300  std::vector<individual_s> parents;
301  std::vector<REAL_> child;
302 
303  // Iteration index
304  static size_t iter_index = 0;
305 
306  // Evaluate fitness of each individual
307  evaluation(population);
308 
309  // Sort the entire population based on fitness
310  sort_population();
311 
312  // Bookkeeping
313  best_individual = sorted_population[sorted_population.size()-1];
314  hfi.push_back(best_individual.fitness); // Best fitness
315  bsf_genome = best_individual.genome; // Best so far genome
316  lfi.push_back(sorted_population[0].fitness); // Lowest fitness
317 
318  // Average fitness across population
319  REAL_ acc = std::accumulate(population.begin(),
320  population.end(),
321  0,
322  average_fitness);
323  fit_avg.push_back(acc / static_cast<REAL_>(mu));
324 
325  bsf.push_back(best_individual.fitness);
326 
327  // Generate new offspring
328  for(size_t i = 0; i < lambda-1; ++i) {
329  // Parents selection
330  // parents = selection(2, 2, false);
331  parents = (this->*selection)(population);
332  parent1 = parents[0];
333  parent2 = parents[1];
334  /* parent1 = parents[i]; */
335  /* parent2 = parents[i+1]; */
336 
337  // Reset the selection flags
338  // This has to be placed outside the loop in case the selection method
339  // is placed outside the loop
340  reset_selection_flags();
341 
342  // Crossover
343  child = (this->*crossover)(parent1.genome, parent2.genome);
344  // child = order_one_crossover(parent1.genome, parent2.genome);
345 
346  // Mutation
347  child = (this->*mutation)(child);
348 
349  // Append the offspring genome list
350  offsprings[i].genome = child;
351  }
352  // Evaluate offspring fitness
353  evaluation(offsprings);
354 
355  // Integrate offspring in the initial population
356  next_generation(replace_perc);
357 
358  // Clip genome
359  clip_genome();
360 
361  // Increase iteration index
362  iter_index++;
363 }
