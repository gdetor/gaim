47 IM::IM(im_parameter_s *im_pms, ga_parameter_s *ga_pms)
48 {
49  size_t num_vertices;
50  a = ga_pms->a; // Lower bound of genome [a, b]
51  b = ga_pms->b; // Upper bound of genome [a, b]
52  num_islands = im_pms->num_islands; // Number of islands
53  num_immigrants = im_pms->num_immigrants; // Number of immigrants
54  migration_interval = im_pms->migration_interval; // Migration interval
55 
56  if (migration_interval > ga_pms->generations) {
57  std::cerr << "Migration interval exceeds number of generations!"
58  << std::endl;
59  exit(-1);
60  }
61 
62  // Compute the migration steps
63  migration_steps = ga_pms->generations;
64 
65  // Read the connectivity graph from a file
66  num_vertices = read_connectivity_graph(im_pms->adj_list_fname);
67  if (num_vertices != num_islands) {
68  std::cerr << "WARNING: Number of islands does not match the number of vertices!"
69  << std::endl;
70  std::cout << "Number of islands has changed to: " << num_vertices
71  << std::endl;
72  num_islands = num_vertices; // Assign number of islands
73  im_pms->num_islands = num_vertices;
74  }
75 
76  // Initialize the GA for each island
77  for (size_t i = 0; i < num_islands; ++i) {
78  island.push_back(GA(ga_pms));
79  }
80 }

115 size_t IM::read_connectivity_graph(std::string fname)
116 {
117  int source, content;
118  size_t num_vertices, num_edges;
119 
120  auto ifile = std::fstream(fname, std::ios::in);
121  if (!ifile) {
122  std::cout << "Unable to open file " << fname << std::endl;
123  exit(1);
124  }
125 
126  ifile >> num_vertices;
127  for (size_t i = 0; i < num_vertices; ++i) {
128  ifile >> source;
129  ifile >> num_edges;
130  for (size_t j = 0; j < num_edges; ++j) {
131  ifile >> content;
132  if (content < 0 && content > (int) num_islands) {
133  std::cerr << "Error: Invalid destination island!" << std::endl;
134  exit(-1);
135  }
136  adj_list[source].push_back(content);
137  }
138  }
139  ifile.close();
140  return num_vertices;
141 }

162 void IM::select_ind2migrate(size_t num_immigrants,
163  size_t unique_id,
164  std::string method)
165 {
166  size_t id;
167  size_t len = island[unique_id].population.size();
168  static std::random_device rd;
169  static std::mt19937 gen(rd());
170  static std::uniform_real_distribution<> probs(a[0], b[0]);
171  std::vector <REAL_> new_genome(island[unique_id].genome_size);
172  std::vector<int> pop(island[unique_id].population.size());
173 
174  std::iota(std::begin(pop), std::end(pop), 0);
175  std::shuffle(std::begin(pop), std::end(pop), gen);
176 
177  mtx.lock();
178  island[unique_id].immigrant.clear();
179  if (method == "random") {
180  for (size_t i = 0; i < num_immigrants; ++i) {
181  id = pop[i];
182  island[unique_id].immigrant.push_back(island[unique_id].sorted_population[id]);
183  }
184  } else if (method == "elit") {
185  for (size_t i = len-1, j = 0; j < num_immigrants; --i, ++j) {
186  island[unique_id].immigrant.push_back(island[unique_id].sorted_population[i]);
187  }
188  } else if (method == "poor") {
189  for (size_t i = 0; i < num_immigrants; ++i) {
190  island[unique_id].immigrant.push_back(island[unique_id].sorted_population[i]);
191  }
192  } else {
193  std::cerr << "No such immigration method exists!" <<std::endl;
194  exit(-1);
195  }
196  mtx.unlock();
197 
198  for (size_t i = 0; i < num_immigrants; ++i) {
199  std::generate(new_genome.begin(), new_genome.end(), [&]{return probs(gen);});
200  island[unique_id].population[island[unique_id].immigrant[i].id].genome = new_genome;
201  island[unique_id].population[island[unique_id].immigrant[i].id].fitness = island[unique_id].fitness(&new_genome[0], new_genome.size());
202  }
203 }

225 void IM::move_immigrants(size_t num_immigrants,
226  size_t unique_id,
227  std::string method="random")
228 {
229  size_t id;
230  static std::random_device rd;
231  static std::mt19937 gen(rd());
232  std::vector<individual_s> tmp(num_immigrants);
233  std::vector<int> pop(island[unique_id].population.size());
234 
235  std::iota(std::begin(pop), std::end(pop), 0);
236  std::shuffle(std::begin(pop), std::end(pop), gen);
237 
238  mtx.lock();
239  for (auto &k : adj_list[unique_id]) {
240  if (method == "random") {
241  size_t i = 0;
242  for (auto &t : island[k].immigrant) {
243  id = pop[i];
244  island[unique_id].population[id].genome = t.genome;
245  island[unique_id].population[id].fitness =
246  island[unique_id].fitness(&t.genome[0], t.genome.size());
247  i++;
248  }
249  } else if (method == "poor") {
250  size_t i = 0;
251  for (auto &t : island[k].immigrant) {
252  island[unique_id].sorted_population[i].genome = t.genome;
253  island[unique_id].sorted_population[i].fitness =
254  island[unique_id].fitness(&t.genome[0], t.genome.size());
255  i++;
256  }
257  island[unique_id].population = island[unique_id].sorted_population;
258  } else if (method == "elit") {
259  id = island[unique_id].population.size();
260  size_t i = 0;
261  for (auto &t : island[k].immigrant) {
262  island[unique_id].sorted_population[id-1-i].genome = t.genome;
263  island[unique_id].sorted_population[id-1-i].fitness =
264  island[unique_id].fitness(&t.genome[0], t.genome.size());
265  i++;
266  }
267  island[unique_id].population = island[unique_id].sorted_population;
268  } else {
269  std::cerr << "No such immigration method exists!" <<std::endl;
270  exit(-1);
271  }
272  }
273  mtx.unlock();
274 }

288 void IM::evolve_island(size_t unique_id,
289  im_parameter_s *im_pms,
290  pr_parameter_s *pr_pms)
291 {
292  std::vector<individual_s> immigrant;
293 
294  island[unique_id].evaluation(island[unique_id].population);
295  for (size_t k = 0; k < migration_steps; ++k) {
296  island[unique_id].run_one_generation();
297  pthread_barrier_wait(&barrier);
298 
299  if (!(k % migration_interval)) {
300  select_ind2migrate(im_pms->num_immigrants,
301  unique_id,
302  im_pms->pick_method);
303  pthread_barrier_wait(&barrier);
304  move_immigrants(im_pms->num_islands,
305  unique_id,
306  im_pms->replace_method);
307  pthread_barrier_wait(&barrier);
308  }
309  }
310  island[unique_id].sort_population();
311  pthread_barrier_wait(&barrier);
312 
313  mtx.lock();
314  if (pr_pms->print_fitness) {
315  print_fitness(island[unique_id].population,
316  unique_id,
317  pr_pms->where2write);
318  }
319  if (pr_pms->print_average_fitness) {
320  print_avg_fitness(island[unique_id].fit_avg,
321  unique_id,
322  pr_pms->where2write);
323  }
324  if (pr_pms->print_bsf) {
325  print_bsf(island[unique_id].bsf, unique_id, pr_pms->where2write);
326  }
327  if (pr_pms->print_best_genome) {
328  print_best_genome(island[unique_id].best_individual.genome,
329  unique_id,
330  pr_pms->where2write);
331  }
332  mtx.unlock();
333 }
