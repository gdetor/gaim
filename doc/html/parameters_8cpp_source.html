40 std::tuple<ga_parameter_s, pr_parameter_s, im_parameter_s> read_parameters_file(const char *fname)
41 {
42  Config cfg;
43  ga_parameter_s tmp;
44  pr_parameter_s print_tmp;
45  im_parameter_s island_tmp;
46 
47  int generations, population_size, num_offsprings, genome_size;
48  int num_replacement, runs;
49  int num_islands, num_immigrants, migration_interval;
50  bool print_fitness, print_avg_fitness, print_bsf, print_best_genome;
51  bool im_enabled;
52  std::string clipping ;
53  int flag;
54  std::string where2write, exp_name, final_path, adj_list_fname;
55  std::string base("./"), rmethod, cmethod;
56  std::string clipping_fname;
57 
58  // Read the file. If there is an error, report it and exit.
59  try
60  {
61  cfg.readFile(fname);
62  }
63  catch(const FileIOException &fioex)
64  {
65  std::cerr << "I/O error while reading file." << std::endl;
66  exit(-1);
67  }
68  catch(const ParseException &pex)
69  {
70  std::cerr << "Parse error at " << pex.getFile() << ":" << pex.getLine()
71  << " - " << pex.getError() << std::endl;
72  exit(-1);
73  }
74 
75  // Get the simulation name.
76  try
77  {
78  std::string name = cfg.lookup("name");
79  exp_name = name;
80  std::cout << "Simulation: " << name << std::endl << std::endl;
81  }
82  catch(const SettingNotFoundException &nfex)
83  {
84  std::cerr << "No 'name' setting in configuration file." << std::endl;
85  }
86 
87  const Setting& root = cfg.getRoot();
88  try
89  {
90  const Setting &ga = root["GA"]["evolution"];
91  const Setting &pr = root["GA"]["print"];
92  const Setting &im = root["GA"]["island_model"];
93 
94  const Setting &a_cfg = ga.lookup("a");
95  std::vector<REAL_> tmp_a;
96  for (int n = 0; n < a_cfg.getLength(); ++n) {
97  tmp_a.push_back(a_cfg[n]);
98  }
99 
100  const Setting &b_cfg = ga.lookup("b");
101  std::vector<REAL_> tmp_b;
102  for (int n = 0; n < b_cfg.getLength(); ++n) {
103  tmp_b.push_back(b_cfg[n]);
104  }
105 
106  // GA parameters
107  if (ga.lookupValue("generations", generations) &&
108  ga.lookupValue("runs", runs) &&
109  ga.lookupValue("population_size", population_size) &&
110  ga.lookupValue("number_of_offsprings", num_offsprings) &&
111  ga.lookupValue("size_of_genome", genome_size) &&
112  ga.lookupValue("clipping", clipping) &&
113  ga.lookupValue("clipping_fname", clipping_fname) &&
114  ga.lookupValue("number_of_replacement", num_replacement)) {
115 
116  if ((generations < 0) || (population_size <= 0) ||
117  (num_offsprings <= 0) || (genome_size <= 0) ||
118  (num_replacement < 0)){
119  std::cerr << "Negative parameters detected!" << std::endl;
120  exit(-1);
121  }
122 
123  tmp.runs = runs;
124  tmp.generations = generations;
125  tmp.population_size = population_size;
126  tmp.num_offsprings = num_offsprings;
127  if (num_offsprings >= population_size) {
128  std::cerr << "Number of offsprings cannot exceed population size!" << std::endl;
129  exit(-1);
130  }
131  tmp.genome_size = genome_size;
132  if (genome_size < 1) {
133  std::cerr << "Genome size cannot be smaller than 1!" << std::endl;
134  exit(-1);
135  }
136  tmp.num_replacement = num_replacement;
137  tmp.clipping = clipping;
138  tmp.clipping_fname = clipping_fname;
139  }
140  tmp.a = tmp_a;
141  tmp.b = tmp_b;
142 
143  // Printing parameters
144  if (pr.lookupValue("print_fitness", print_fitness) &&
145  pr.lookupValue("print_average_fitness", print_avg_fitness) &&
146  pr.lookupValue("print_bsf", print_bsf) &&
147  pr.lookupValue("print_best_genome", print_best_genome) &&
148  pr.lookupValue("where_to_write", where2write)) {
149 
150  print_tmp.print_fitness = print_fitness;
151  print_tmp.print_average_fitness = print_avg_fitness;
152  print_tmp.print_bsf = print_bsf;
153  print_tmp.print_best_genome = print_best_genome;
154  print_tmp.where2write = where2write;
155  print_tmp.experiment_name = exp_name;
156 
157  }
158 
159  // IM parameters
160  if (im.lookupValue("im_enabled", im_enabled) &&
161  im.lookupValue("number_of_immigrants", num_immigrants) &&
162  im.lookupValue("migration_interval", migration_interval) &&
163  im.lookupValue("number_of_islands", num_islands) &&
164  im.lookupValue("island_model_graph_file", adj_list_fname) &&
165  im.lookupValue("candidate_immigrants_replace_method", rmethod) &&
166  im.lookupValue("candidate_immigrants_pickup_method", cmethod)) {
167 
168  island_tmp.num_immigrants = num_immigrants;
169  island_tmp.num_islands = num_islands;
170  island_tmp.migration_interval = migration_interval;
171  island_tmp.pick_method = cmethod;
172  island_tmp.replace_method = rmethod;
173  island_tmp.is_im_enabled = im_enabled;
174  island_tmp.adj_list_fname = adj_list_fname;
175  }
176 
177  if (island_tmp.is_im_enabled && tmp.runs > 1) {
178  std::cerr << "Is not allowed to use more than 1 runs when IM is enabled!" << std::endl;
179  exit(-1);
180  }
181 
182  if (island_tmp.is_im_enabled) {
183  if (print_tmp.where2write == "stdout") {
184  print_tmp.where2write = "./results";
185  }
186  }
187 
188  if (print_tmp.where2write != "stdout") {
189  if (!is_path_exist(print_tmp.where2write)) {
190  flag = mkdir_(print_tmp.where2write);
191  if (flag) {
192  exit(-1);
193  }
194  }
195  }
196  }
197  catch(const SettingNotFoundException &nfex)
198  {
199  //Ignore
200  }
201  return std::make_tuple(tmp, print_tmp, island_tmp);
202 }
