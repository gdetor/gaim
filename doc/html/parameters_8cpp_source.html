42 std::tuple<ga_parameter_s,
43  pr_parameter_s,
44  im_parameter_s> read_parameters_file(const char *fname)
45 {
46  Config cfg;
47  ga_parameter_s tmp;
48  pr_parameter_s print_tmp;
49  im_parameter_s island_tmp;
50 
51  int generations, population_size, num_offsprings, genome_size, k;
52  int num_replacement, runs;
53  int num_islands, num_immigrants, migration_interval;
54  int flag, time, order;
55  int num_parents, lower_bound;
56  REAL_ bias, mutation_rate, variance;
57  REAL_ low_bound, up_bound;
58  bool print_fitness, print_avg_fitness, print_bsf, print_best_genome;
59  bool im_enabled, replace, is_real;
60  std::string clipping, method;
61  std::string where2write, exp_name, final_path, adj_list_fname;
62  std::string base("./"), rmethod, cmethod;
63  std::string clipping_fname;
64 
65  // Read the file. If there is an error, report it and exit.
66  try
67  {
68  cfg.readFile(fname);
69  }
70  catch(const FileIOException &fioex)
71  {
72  std::cerr << "I/O error while reading file." << std::endl;
73  exit(-1);
74  }
75  catch(const ParseException &pex)
76  {
77  std::cerr << "Parse error at " << pex.getFile() << ":" << pex.getLine()
78  << " - " << pex.getError() << std::endl;
79  exit(-1);
80  }
81 
82  // Get the simulation name.
83  try
84  {
85  std::string name = cfg.lookup("name");
86  exp_name = name;
87  std::cout << "Simulation: " << name << std::endl << std::endl;
88  }
89  catch(const SettingNotFoundException &nfex)
90  {
91  std::cerr << "No 'name' setting in configuration file." << std::endl;
92  }
93 
94  const Setting& root = cfg.getRoot();
95  try
96  {
97  const Setting &ga = root["GA"]["evolution"];
98  const Setting &sel = root["GA"]["selection"];
99  const Setting &cross = root["GA"]["crossover"];
100  const Setting &mut = root["GA"]["mutation"];
101  const Setting &pr = root["GA"]["print"];
102  const Setting &im = root["GA"]["island_model"];
103 
104  const Setting &a_cfg = ga.lookup("a");
105  std::vector<REAL_> tmp_a;
106  for (int n = 0; n < a_cfg.getLength(); ++n) {
107  tmp_a.push_back(a_cfg[n]);
108  }
109 
110  const Setting &b_cfg = ga.lookup("b");
111  std::vector<REAL_> tmp_b;
112  for (int n = 0; n < b_cfg.getLength(); ++n) {
113  tmp_b.push_back(b_cfg[n]);
114  }
115 
116  // GA parameters
117  if (ga.lookupValue("generations", generations) &&
118  ga.lookupValue("runs", runs) &&
119  ga.lookupValue("population_size", population_size) &&
120  ga.lookupValue("number_of_offsprings", num_offsprings) &&
121  ga.lookupValue("size_of_genome", genome_size) &&
122  ga.lookupValue("clipping", clipping) &&
123  ga.lookupValue("clipping_fname", clipping_fname) &&
124  ga.lookupValue("number_of_replacement", num_replacement)) {
125 
126  if ((generations < 0) || (population_size <= 0) ||
127  (num_offsprings <= 0) || (genome_size <= 0) ||
128  (num_replacement < 0)){
129  std::cerr << "Negative parameters detected!" << std::endl;
130  exit(-1);
131  }
132 
133  tmp.runs = runs;
134  tmp.generations = generations;
135  tmp.population_size = population_size;
136  tmp.num_offsprings = num_offsprings;
137  if (num_offsprings >= population_size) {
138  std::cerr << "Number of offsprings cannot exceed population size!" << std::endl;
139  exit(-1);
140  }
141  tmp.genome_size = genome_size;
142  if (genome_size < 1) {
143  std::cerr << "Genome size cannot be smaller than 1!" << std::endl;
144  exit(-1);
145  }
146  tmp.num_replacement = num_replacement;
147  tmp.clipping = clipping;
148  tmp.clipping_fname = clipping_fname;
149  }
150  tmp.a = tmp_a;
151  tmp.b = tmp_b;
152 
153  // Selection parameters
154  if (sel.lookupValue("selection_method", method) &&
155  sel.lookupValue("bias", bias) &&
156  sel.lookupValue("num_parents", num_parents) &&
157  sel.lookupValue("lower_bound", lower_bound) &&
158  sel.lookupValue("k", k) &&
159  sel.lookupValue("replace", replace)) {
160 
161  tmp.sel_pms.selection_method = method;
162  tmp.sel_pms.bias = bias;
163  tmp.sel_pms.num_parents = num_parents;
164  tmp.sel_pms.lower_bound = lower_bound;
165  tmp.sel_pms.k = k;
166  tmp.sel_pms.replace = replace;
167  }
168 
169  // Crossover parameters
170  if (cross.lookupValue("crossover_method", method)) {
171  tmp.cross_pms.crossover_method = method;
172  }
173 
174  // Selection parameters
175  if (mut.lookupValue("mutation_method", method) &&
176  mut.lookupValue("mutation_rate", mutation_rate) &&
177  mut.lookupValue("variance", variance) &&
178  mut.lookupValue("low_bound", low_bound) &&
179  mut.lookupValue("up_bound", up_bound) &&
180  mut.lookupValue("time", time) &&
181  mut.lookupValue("order", order) &&
182  mut.lookupValue("is_real", is_real)) {
183 
184  tmp.mut_pms.mutation_method = method;
185  tmp.mut_pms.mutation_rate = mutation_rate;
186  tmp.mut_pms.variance = variance;
187  tmp.mut_pms.low_bound = low_bound;
188  tmp.mut_pms.up_bound = up_bound;
189  tmp.mut_pms.time = time;
190  tmp.mut_pms.order = order;
191  tmp.mut_pms.is_real = is_real;
192  }
193 
194  // Printing parameters
195  if (pr.lookupValue("print_fitness", print_fitness) &&
196  pr.lookupValue("print_average_fitness", print_avg_fitness) &&
197  pr.lookupValue("print_bsf", print_bsf) &&
198  pr.lookupValue("print_best_genome", print_best_genome) &&
199  pr.lookupValue("where_to_write", where2write)) {
200 
201  print_tmp.print_fitness = print_fitness;
202  print_tmp.print_average_fitness = print_avg_fitness;
203  print_tmp.print_bsf = print_bsf;
204  print_tmp.print_best_genome = print_best_genome;
205  print_tmp.where2write = where2write;
206  print_tmp.experiment_name = exp_name;
207 
208  }
209 
210  // IM parameters
211  if (im.lookupValue("im_enabled", im_enabled) &&
212  im.lookupValue("number_of_immigrants", num_immigrants) &&
213  im.lookupValue("migration_interval", migration_interval) &&
214  im.lookupValue("number_of_islands", num_islands) &&
215  im.lookupValue("island_model_graph_file", adj_list_fname) &&
216  im.lookupValue("candidate_immigrants_replace_method", rmethod) &&
217  im.lookupValue("candidate_immigrants_pickup_method", cmethod)) {
218 
219  island_tmp.num_immigrants = num_immigrants;
220  island_tmp.num_islands = num_islands;
221  island_tmp.migration_interval = migration_interval;
222  island_tmp.pick_method = cmethod;
223  island_tmp.replace_method = rmethod;
224  island_tmp.is_im_enabled = im_enabled;
225  island_tmp.adj_list_fname = adj_list_fname;
226  }
227 
228  if (island_tmp.is_im_enabled && tmp.runs > 1) {
229  std::cerr << "Is not allowed to use more than 1 runs when IM is enabled!" << std::endl;
230  exit(-1);
231  }
232 
233  if (island_tmp.is_im_enabled) {
234  if (print_tmp.where2write == "stdout") {
235  print_tmp.where2write = "./results";
236  }
237  }
238 
239  if (print_tmp.where2write != "stdout") {
240  if (!is_path_exist(print_tmp.where2write)) {
241  flag = mkdir_(print_tmp.where2write);
242  if (flag) {
243  exit(-1);
244  }
245  }
246  }
247  }
248  catch(const SettingNotFoundException &nfex)
249  {
250  //Ignore
251  }
252  return std::make_tuple(tmp, print_tmp, island_tmp);
253 }
