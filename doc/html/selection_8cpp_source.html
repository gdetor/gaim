192 std::vector<individual_s> GA::linear_rank_selection(std::vector<individual_s> &population) {
193  size_t index = 0;
194  REAL_ prob = 0;
195  std::vector<individual_s> selected_individuals;
196  individual_s best;
197  static std::random_device rd;
198  static std::mt19937 gen(rd());
199  static std::uniform_real_distribution<> uniform(0, 1);
200 
201  // Check if the number of parents is greater than the population size
202  if (population.size() < num_parents) {
203  std::cout << "WARNING: Rank Selection Method" << std::endl;
204  std::cout << "Parents size > Population Size" << std::endl;
205  std::cout << "Parents size reduced to population size" << std::endl;
206  num_parents = population.size();
207  }
208 
209  // If number of parents is the same as the population size (generational
210  // replacement)
211  if (num_parents == population.size()) {
212  selected_individuals = population;
213  return selected_individuals;
214  }
215 
216  // Sort the individuals based on their fitness - Create the rank
217  std::sort(population.begin(), population.end(), compare_fitness);
218 
219  // Select the individuals from the population without replacement
220  if (replace == false) {
221  while(selected_individuals.size() != num_parents) {
222  REAL_ r = uniform(gen);
223  for (size_t i = 0; i < population.size(); ++i) {
224  prob += (REAL_) (i+1) / population.size();
225  if ((r < prob) && (population[i].is_selected == false)) {
226  index = i;
227  break;
228  }
229  }
230  selected_individuals.push_back(population[index]);
231  population[index].is_selected = true;
232  prob = 0;
233  }
234  // Select the individuals from the population with replacement
235  } else {
236  for (size_t k = 0; k < num_parents; ++k) {
237  REAL_ r = uniform(gen);
238  for (size_t i = 0; i < population.size(); ++i) {
239  prob += (REAL_) (i+1) / population.size();
240  if (r < prob) {
241  best = population[i];
242  break;
243  }
244  }
245  selected_individuals.push_back(best);
246  prob = 0;
247  }
248  }
249  return selected_individuals;
250 }

327

342  index = j;

396  selected_individuals = population;
397  return selected_individuals;
398  }
399 
400  // Select the individuals from the population without replacement
401  if (replace == false) {
402  max_fitness = maximum_fitness(population);
403  std::vector<size_t> indices(population.size());
404  std::iota(indices.begin(), indices.end(), 0);
405  for(size_t i = 0; i < num_parents; ++i) {
406  r = float_random(0, 1);
407  index = indices[int_random(0, indices.size()-1)];
408  prob = population[index].fitness / max_fitness;
409  if ((prob > r) && (population[index].is_selected == false)) {
410  selected_individuals.push_back(population[index]);
411  population[index].is_selected = true;
412  remove_at(indices, index);
413  }
414  max_fitness = nonselected_maximum_fitness(population);
415  }
416  // Select the individuals from the population with replacement
417  } else {
418  for(size_t i = 0; i < num_parents; ++i) {

421  r = float_random(0, 1);

443 REAL_ calculate_whitley_factor(REAL_ bias) {
444  REAL_ tmp;
445  REAL_ R = float_random(0, 1);
446  tmp = (bias - sqrt(bias*bias - 4.0*(bias-1) * R)) / 2.0 / (bias-1);
447  return tmp;
