103 ga_results_s return_best_results(std::vector<GA> population,
104  std::string return_type) {
105  int best_index(0);
106  std::vector<REAL_> norms;
107  ga_results_s res;
108 
109  norms = compute_population_norms(population);
110 
111  if (return_type == "minimum") {
112  best_index = argmin(norms);
113  }else if (return_type == "random") {
114  best_index = (int) int_random(0, norms.size()-1);
115  } else {
116  best_index = argmax(norms);
117  }
118  res.bsf = population[best_index].get_bsf();
119  res.average_fitness = population[best_index].get_average_fitness();
120  res.genome = population[best_index].get_best_genome();
121  return res;
122 }
